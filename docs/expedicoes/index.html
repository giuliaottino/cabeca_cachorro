find_up <- function(rel_path, start = getwd(), max_levels = 8) {
  d <- normalizePath(start, winslash = "/", mustWork = FALSE)
  for (i in 0:max_levels) {
    cand <- file.path(d, rel_path)
    if (file.exists(cand)) return(cand)
    parent <- dirname(d)
    if (identical(parent, d)) break
    d <- parent
  }
  NA_character_
}

pick_col <- function(data, candidates, required = TRUE, label = NULL) {
  hit <- intersect(candidates, names(data))
  if (length(hit) > 0) return(hit[1])
  if (required) {
    stop(
      "Não encontrei a coluna ", ifelse(is.null(label), paste(candidates, collapse = "/"), label), ".\n",
      "Colunas disponíveis no input: ", paste(names(data), collapse = ", ")
    )
  }
  NA_character_
}

to_num <- function(x) {
  if (is.numeric(x)) return(x)
  suppressWarnings(as.numeric(gsub(",", ".", trimws(as.character(x)))))
}

nz <- function(x) {

safe_html <- function(x) {
  x <- as.character(x)
  x[is.na(x) | trimws(x) == ""] <- "—"
  htmltools::htmlEscape(x)
}

slugify <- function(x) {
  x <- as.character(x)
  x[is.na(x) | trimws(x) == ""] <- "expedicao"
  x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT")
  x <- tolower(x)
  x <- gsub("[^a-z0-9]+", "-", x)
  x <- gsub("(^-+|-+$)", "", x)
  x[x == ""] <- "expedicao"
  make.unique(x, sep = "-")
}

library(htmltools)

photo_root_fs_rel <- file.path("_input_data", "expedicoes_fotos")
photo_root_html_rel <- file.path("..", "_input_data", "expedicoes_fotos")
photo_root_abs <- find_up(photo_root_fs_rel)

get_photos <- function(folder) {
  if (is.na(photo_root_abs) || is.na(folder) || !nzchar(trimws(folder))) {
    return(character())
  }

  d <- file.path(photo_root_abs, folder)

  if (!dir.exists(d)) {
    return(character())
  }

  f <- list.files(
    d,
    pattern = "\\.(jpg|jpeg|png|webp)$",
    full.names = FALSE,
    ignore.case = TRUE
  )

  f <- f[!grepl("^\\.", f)]

  if (length(f) == 0) {
    return(character())
  }

  gsub("\\\\", "/", file.path(photo_root_html_rel, folder, f))
}

make_meta <- function(label, value, key = NULL) {
  if (!nz(value)) return(NULL)

  if (!is.null(key) && nzchar(key)) {
    tags$span(
      class = "meta-pill",
      tags$span(`data-i18n` = key, label),
      ": ",
      value
    )
  } else {
    tags$span(
      class = "meta-pill",
      paste0(label, ": ", value)
    )
  }
}

make_exp_card <- function(i) {
  x <- expedicoes[i, , drop = FALSE]

  photos <- get_photos(x$pasta_fotos)

  title <- ifelse(
    nz(x$expedicao),
    x$expedicao,
    paste0("Expedição ", i)
  )

  summary_txt <- ifelse(
    nz(x$resumo),
    x$resumo,
    x$observacoes
  )

  gallery <- if (length(photos) > 0) {
    tags$div(
      class = "exp-carousel",
      lapply(seq_along(photos), function(j) {
        tags$figure(
          tags$img(
            src = photos[j],
            alt = paste(title, "foto", j),
            loading = "lazy",
            decoding = "async"
          ),
          tags$figcaption(
            tagList(tags$span(`data-i18n` = "common.photo", "Foto"), " ", j, " · ", title)
          )
        )
      })
    )
  } else {
    tags$div(
      class = "empty-gallery",
      tagList(
        tags$span(`data-i18n` = "expeditions.card.noPhoto", "Nenhuma foto encontrada para esta expedição."),
        " ",
        tags$span(`data-i18n` = "expeditions.card.expectedFolder", "Pasta esperada:"),
        " _input_data/expedicoes_fotos/",
        x$pasta_fotos
      )
    )
  }

  tags$section(
    class = "expedition-card",
    id = x$section_id,
    tags$div(class = "expedition-eyebrow", `data-i18n` = "expeditions.card.eyebrow", "Expedição"),
    tags$h3(title),
    tags$div(
      class = "expedition-meta",
      make_meta("Data", x$data_exp, "common.date"),
      make_meta("Localidade", x$localidade, "common.locality"),
      make_meta("Comunidade", x$comunidade, "common.community"),
      make_meta("Município", x$municipio, "common.municipality"),
      make_meta("Equipe", x$equipe, "common.team")
    ),
    if (nz(summary_txt)) {
      tags$p(class = "expedition-summary", summary_txt)
    },
    gallery
  )
}

if (nrow(expedicoes) > 0) {
  tagList(
    tags$div(
      class = "expedition-grid",
      lapply(seq_len(nrow(expedicoes)), make_exp_card)
    )
  )
} else {
  tags$p(`data-i18n` = "expeditions.list.empty", "Nenhuma expedição cadastrada.")
}
