find_up <- function(rel_path, start = getwd(), max_levels = 8) {
  d <- normalizePath(start, winslash = "/", mustWork = FALSE)
  for (i in 0:max_levels) {
    cand <- file.path(d, rel_path)
    if (file.exists(cand)) return(cand)
    parent <- dirname(d)
    if (identical(parent, d)) break
    d <- parent
  }
  NA_character_
}

nz <- function(x) {
  !is.na(x) & nzchar(trimws(as.character(x)))
}

safe_html <- function(x) {
  x <- as.character(x)
  x[is.na(x) | trimws(x) == ""] <- "—"
  htmltools::htmlEscape(x)
}

slugify <- function(x) {
  x <- as.character(x)
  x[is.na(x) | trimws(x) == ""] <- "membro"
  x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT")
  x <- tolower(x)
  x <- gsub("[^a-z0-9]+", "-", x)
  x <- gsub("(^-+|-+$)", "", x)
  x[x == ""] <- "membro"
  make.unique(x, sep = "-")
}

normalize_name <- function(x) {
  x <- as.character(x)
  x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT")
  x <- tolower(x)
  x <- gsub("[^a-z0-9]+", " ", x)
  x <- trimws(gsub("\\s+", " ", x))
  x
}

normalize_col_id <- function(x) {
  x <- as.character(x)
  x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT")
  x <- tolower(x)
  x <- gsub("[^a-z0-9]+", "", x)
  x
}

get_col <- function(data, candidates, required = TRUE, label = NULL) {
  hit <- intersect(candidates, names(data))
  if (length(hit) > 0) return(hit[1])

  data_names_norm <- normalize_col_id(names(data))
  candidates_norm <- normalize_col_id(candidates)
  hit_norm <- which(data_names_norm %in% candidates_norm)
  if (length(hit_norm) > 0) return(names(data)[hit_norm[1]])

  if (required) {
    stop(
      "Não encontrei a coluna ", ifelse(is.null(label), paste(candidates, collapse = "/"), label), ".\n",
      "Colunas disponíveis no input: ", paste(names(data), collapse = ", ")
    )
  }
  NA_character_
}

get_value <- function(row, possible_names, default = "") {
  for (name in possible_names) {
    if (name %in% names(row)) {
      value <- row[[name]][1]
      if (!is.na(value) && nzchar(trimws(as.character(value)))) {
        return(as.character(value))
      }
    }
  }

  row_names_norm <- normalize_col_id(names(row))
  possible_norm <- normalize_col_id(possible_names)
  hit <- which(row_names_norm %in% possible_norm)

  if (length(hit) > 0) {
    value <- row[[hit[1]]][1]
    if (!is.na(value) && nzchar(trimws(as.character(value)))) {
      return(as.character(value))
    }
  }

  default
}

make_photo_filename <- function(member_name) {
  x <- tolower(member_name)
  x <- iconv(x, from = "UTF-8", to = "ASCII//TRANSLIT")
  x <- gsub("[^a-z0-9]+", "_", x)
  x <- gsub("_+", "_", x)
  x <- gsub("^_|_$", "", x)
  paste0(x, ".jpg")
}

make_icon <- function(href, icon_file, alt, is_email = FALSE) {
  if (!nz(href)) return("")

  href_safe <- htmlEscape(href)
  icon_src <- paste0("../_input_data/icons/", icon_file)

  if (is_email) {
    paste0(
      '<a class="icon-link" href="mailto:', href_safe, '" aria-label="', alt, '">',
      '<img src="', icon_src, '" alt="', alt, '"/>',
      '</a>'
    )
  } else {
    paste0(
      '<a class="icon-link" href="', href_safe, '" target="_blank" rel="noopener" aria-label="', alt, '">',
      '<img src="', icon_src, '" alt="', alt, '"/>',
      '</a>'
    )
  }
}

generate_member_card <- function(member) {
  email <- get_value(member, c("Email", "email", "E.mail", "E.mail."))

  orcid <- get_value(member, c("ORCID", "Orcid", "orcid", "ORCID."))

    make_icon(scholar, "_scholar.png", "Google Scholar"),

      '<p class="team-icons">',
      icon_html,
      '</p>'
    )

    '<p class="team-icons team-icons-empty" aria-hidden="true"></p>'
  }

    '<article class="team-card" id="', htmlEscape(member$member_slug[1]),

<script>
document.addEventListener("DOMContentLoaded", function () {
  const searchInput = document.getElementById("team-search");
  const filterButtons = document.querySelectorAll(".team-filter");
  const cards = document.querySelectorAll(".team-card");
  const sections = document.querySelectorAll(".team-section");
  let activeFilter = "all";

  function normalizeText(value) {
    return (value || "")
      .toLowerCase()
      .normalize("NFD")
      .replace(/[\u0300-\u036f]/g, "")
      .replace(/[^a-z0-9]+/g, " ")
      .trim();
  }

  function applyFilters() {
    const term = normalizeText(searchInput ? searchInput.value : "");

    cards.forEach(function (card) {
      const role = card.dataset.role || "";
      const haystack = normalizeText(
        (card.dataset.search || "") + " " +
        (card.dataset.name || "") + " " +
        (card.dataset.institution || "") + " " +
        (card.dataset.projectRole || "") + " " +
        (card.textContent || "")
      );
      const words = term.split(/\s+/).filter(Boolean);
      const matchesText = words.length === 0 || words.every(word => haystack.includes(word));
      const matchesRole = activeFilter === "all" || role === activeFilter;

      card.hidden = !(matchesText && matchesRole);
    });

    sections.forEach(function (section) {
      const visibleCards = Array.from(section.querySelectorAll(".team-card")).filter(function (card) {
        return !card.hidden;
      });
      const empty = section.querySelector(".empty-team-message");
      const grid = section.querySelector(".team-grid");

      if (empty) empty.style.display = visibleCards.length ? "none" : "block";
      if (grid) grid.hidden = !visibleCards.length;
    });
  }

  if (searchInput) {
    searchInput.addEventListener("input", applyFilters);
  }

  filterButtons.forEach(function (button) {
    button.addEventListener("click", function () {
      activeFilter = button.dataset.filter || "all";
      filterButtons.forEach(btn => btn.classList.remove("active"));
      button.classList.add("active");
      applyFilters();
    });
  });

  applyFilters();
});
</script>
